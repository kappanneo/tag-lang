## tag01

### tag01-textual

```
[0] :: [01]                         -- flse
[0] : []

[1] :: [01]                         -- true

[00] :: ( [|>]:[01] )               -- null? - (=? null)
[00] : ([]:[1] [0])

[01] :: |>                          -- bool
[01] : < [0] || [1] >

[10] :: ( [01]:[01] )               -- flse? - (=? flse)
[10] : [00]

[11] :: ( [01]:[01] )               -- true? - (=? true)
[11] : ( [0]:[0] [1]:[1] )

[000] :: ( [01]:[111] )             -- pure
[000] : ( [0]:([010] []) [1]:([101] []) )

[001] :: ( [111]:[01] )             -- last
[001] :
 ( []:[]
    ( [([010] [])] : [0]
       ( [([101] [])] : [1]
          ( [([010] |[0000]>)] : ([001] [0000])
            [([101] |[0000]>)] : ([001] [0000])
          )
       )
    )
 )

[010] :: ( [111]:[111] )            -- cons 0

[011] :: ( [111]:[111] )            -- tail
[011] : 
 ( []:[]
    ( [([010] |[0000]>)] : [0000]
      [([101] |[0000]>)] : [0000]
    )
 )

[100] :: ( [111]:[01] )             -- head
[100] :
 ( []:[]
    ( [([010] |>)] : [0]
      [([101] |>)] : [1]
    )
 )

[101] :: ( [111]:[111] )            -- cons 1

[110] :: ( [111]:[111] )            -- init
[110] :
 ( []:[]
    ( [([010] [])] : [0]
       ( [([101] [])] : [1]
          ( [([010] |[0000]>)] : ([010] ([110] [0000]))
            [([101] |[0000]>)] : ([101] ([110] [0000]))
          )
       )
    )
 )

[111] :: |>                         -- list bool
[111] : < [] || < ([010] |[1111]>) || ([101] |[1111]>) > >
```
#### possible extention
```
[0000] :: []                         -- just 0

[0001] :: []                         -- just 1

[0010] :: |>                         -- mayb bool
[0010] : < [] || < [0000] || [0001] >  >

[0100] :: ()                         -- mbhd
[0100] : 
 ( [] : []
    ( [([010] |>)] : [0]
      [([101] |>)] : [1]
    )
 )

[1111] :: ( [|00000>]:[|00000>] )    -- same
[1111] : ( [(|[00000]>)] : [00000] )
```
from haskell
```
-- forall a. = [(|[00000]>)] :
-- ( | )     = < || >
```
